## ✅ 확정된 Q&A 결정사항 (14개)

1. **랜딩페이지**: `/` 루트, 로그인은 `/login`
2. **Fallback 영상**: 다한힘 소개 영상 1개 (관리자 페이지에서 관리)
3. **광고 5개 미만**: fallback으로 채워 항상 5개 유지
4. **10초 영상**: 직접 업로드 + 플랫폼 편집기(10초 구간+텍스트, 수수료) 둘 다 지원
5. **결제**: PG 온라인 + 계좌이체(웹훅 자동처리) 둘 다
6. **모바일 광고 슬롯**: 프리미엄 배너 위에 추가
7. **스폰서드 포스트**: 나노바나나 이미지, 매니저/스태프 다른 광고
8. **모바일**: 웹 반응형 (별도 앱 없음)
9. **SMS**: 솔라피 = 카카오 알림톡, 알리고 = SMS fallback
10. **전자계약**: 엔터↔방송국 먼저, 확장 가능 템플릿 구조
11. **관리자 페이지**: 4종 전부 (슈퍼/협회/매니지먼트/방송국)
12. **강제 시청 영상**: ad_videos 활성 영상 랜덤
13. **채팅 첨부**: 이미지+문서+영상 100MB, 관리자 설정 가능
14. **추가 회의록**: 투표/공지만 포함, 일감 매칭은 다음 단계

ALTER TABLE ad_videos ENABLE ROW LEVEL SECURITY;

-- 매니저: 본인 광고만 조회/수정
CREATE POLICY "manager_own" ON ad_videos
  FOR ALL USING (manager_id = auth.uid());

-- 관리자: 전체 접근
CREATE POLICY "admin_all" ON ad_videos
  FOR ALL USING (
    EXISTS (SELECT 1 FROM profiles WHERE id = auth.uid() AND role = 'admin')

CREATE POLICY "public_active" ON ad_videos
  FOR SELECT USING (
    status = 'active'
    AND ad_type = 'idol_video'
    AND ad_start <= CURRENT_DATE
    AND ad_end >= CURRENT_DATE
  );

-- updated_at 자동 갱신 트리거
CREATE OR REPLACE FUNCTION update_updated_at()
RETURNS TRIGGER AS $$
BEGIN NEW.updated_at = NOW(); RETURN NEW; END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER ad_videos_updated_at
  BEFORE UPDATE ON ad_videos
  FOR EACH ROW EXECUTE FUNCTION update_updated_at();

interface VideoItem {

  title: string;

export function useShuffledVideos(count = 5): VideoItem[] {
  const [videos, setVideos] = useState<VideoItem[]>([]);
  const supabase = createClientComponentClient();

  useEffect(() => {
    async function fetchVideos() {
      // 활성 광고 영상
      const { data: ads } = await supabase
        .from('ad_videos')
        .select('id, video_url, youtube_url, youtube_video_id, thumbnail_url, title')
        .eq('status', 'active')
        .eq('ad_type', 'idol_video')
        .lte('ad_start', new Date().toISOString().split('T')[0])
        .gte('ad_end', new Date().toISOString().split('T')[0]);

      let pool = [...(ads ?? [])];

      // fallback으로 채우기
      if (pool.length < count) {
        const { data: fallbacks } = await supabase
          .from('fallback_videos')
          .select('id, video_url, youtube_url, youtube_video_id, thumbnail_url, title')
          .eq('is_active', true)
          .order('display_order');
        pool = [...pool, ...(fallbacks ?? [])];
      }

      // 셔플
      const shuffled = pool.sort(() => Math.random() - 0.5).slice(0, count);
      setVideos(shuffled);
    }

    fetchVideos();
  }, []);

  return videos;
}

// hooks/useAdSlider.ts
import { useEffect, useRef, useState } from 'react';

export function useAdSlider(total: number, intervalMs = 10000) {
  const [current, setCurrent] = useState(0);
  const timer = useRef<NodeJS.Timeout>();

  useEffect(() => {
    if (total === 0) return;
    timer.current = setInterval(() => {
      setCurrent(prev => (prev + 1) % total);
    }, intervalMs);
    return () => clearInterval(timer.current);
  }, [total, intervalMs]);

  return { current, setCurrent };
}

// components/chat/FileAttachment.tsx
'use client';
import { useRef, useState } from 'react';
import { createClientComponentClient } from '@supabase/auth-helpers-nextjs';

const ALLOWED_TYPES: Record<string, string[]> = {
  image: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
  document: ['application/pdf', 'application/msword',
             'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'],
  video: ['video/mp4', 'video/quicktime'],
};

export default function FileAttachment({ onUpload }: { onUpload: (url: string, type: string, name: string) => void }) {
  const inputRef = useRef<HTMLInputElement>(null);
  const [uploading, setUploading] = useState(false);
  const supabase = createClientComponentClient();

  const handleFile = async (file: File) => {
    // 설정에서 최대 파일 크기 가져오기
    const { data: setting } = await supabase
      .from('system_settings')
      .select('value')
      .eq('key', 'chat_max_file_size_mb')
      .single();
    const maxMB = Number(setting?.value ?? 100);

    if (file.size > maxMB * 1024 * 1024) {
      alert(`파일 크기는 ${maxMB}MB 이하여야 합니다.`);
      return;
    }

    setUploading(true);
    const path = `${Date.now()}_${file.name}`;
    const { data, error } = await supabase.storage
      .from('chat-attachments')
      .upload(path, file);

    if (!error && data) {
      const { data: { publicUrl } } = supabase.storage
        .from('chat-attachments')
        .getPublicUrl(data.path);
      onUpload(publicUrl, file.type, file.name);
    }
    setUploading(false);
  };

  return (
    <>
      <input
        ref={inputRef} type="file" className="hidden"
        accept={Object.values(ALLOWED_TYPES).flat().join(',')}
        onChange={e => e.target.files?.[0] && handleFile(e.target.files[0])}
      />
      <button onClick={() => inputRef.current?.click()} disabled={uploading}
        className="p-2 text-gray-500 hover:text-[#374A67]">
        📎
      </button>
    </>
  );
}

// components/contracts/SignatureModal.tsx
// Canvas API로 서명 입력
// 서명 완료 → signed_at_a 또는 signed_at_b 업데이트
// 양측 서명 완료 → status: 'completed'

export default function SignatureModal({ contractId, party }: { contractId: string; party: 'a' | 'b' }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [signed, setSigned] = useState(false);

  const handleSign = async () => {
    const signatureDataUrl = canvasRef.current?.toDataURL();
    const updateField = party === 'a' ? 'signed_at_a' : 'signed_at_b';

    await supabase.from('contracts').update({
      [updateField]: new Date().toISOString(),
      status: party === 'b' ? 'completed' : 'signed_a',
    }).eq('id', contractId);

    setSigned(true);
  };

  return (
    <div>
      <canvas ref={canvasRef} width={400} height={150}
        className="border rounded bg-white touch-none" />
      <button onClick={handleSign}>서명 완료</button>
    </div>
  );
}
